package com.solveria.core.workforce.domain.model.vo;

import lombok.AllArgsConstructor;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.NoArgsConstructor;

import javax.persistence.Embeddable;
import java.io.Serializable;

/**
 * Value Object: ContactPoint
 *
 * Representa un punto de contacto (email, teléfono, dirección).
 * Es reemplazable; si cambia, se reemplaza completamente (no tiene identidad propia).
 *
 * Invariantes:
 * - Email debe tener formato válido si está presente
 * - Teléfono debe ser formato válido (opcional)
 * - Dirección es de libre formato (opcional)
 */
@Embeddable
@Getter
@NoArgsConstructor
@AllArgsConstructor
@EqualsAndHashCode
public class ContactPoint implements Serializable {

    private static final long serialVersionUID = 1L;
    private static final String EMAIL_PATTERN = "^[A-Za-z0-9+_.-]+@(.+)$";

    private String email;
    private String phone;
    private String address;

    public static ContactPoint create(String email, String phone, String address) {
        if (email != null && !email.matches(EMAIL_PATTERN)) {
            throw new IllegalArgumentException("Email format inválido");
        }
        return new ContactPoint(email, phone, address);
    }

    public ContactPoint updateEmail(String newEmail) {
        if (newEmail != null && !newEmail.matches(EMAIL_PATTERN)) {
            throw new IllegalArgumentException("Email format inválido");
        }
        return new ContactPoint(newEmail, this.phone, this.address);
    }

    public ContactPoint updatePhone(String newPhone) {
        return new ContactPoint(this.email, newPhone, this.address);
    }

    public ContactPoint updateAddress(String newAddress) {
        return new ContactPoint(this.email, this.phone, newAddress);
    }
}

package com.solveria.core.workforce.domain.model.vo;

import lombok.AllArgsConstructor;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.NoArgsConstructor;

import javax.persistence.Embeddable;
import java.io.Serializable;

/**
 * Value Object: HeadcountPlan
 *
 * Plan de ocupación de plazas.
 *
 * Invariante: No se puede exceder el límite de slots.
 */
@Embeddable
@Getter
@NoArgsConstructor
@AllArgsConstructor
@EqualsAndHashCode
public class HeadcountPlan implements Serializable {

    private static final long serialVersionUID = 1L;

    private Integer maxSlots;
    private Integer currentSlots;

    public static HeadcountPlan create(Integer maxSlots) {
        if (maxSlots == null || maxSlots <= 0) {
            throw new IllegalArgumentException("maxSlots debe ser mayor a 0");
        }
        return new HeadcountPlan(maxSlots, 0);
    }

    public boolean hasVacancy() {
        return currentSlots < maxSlots;
    }

    public void occupy() {
        if (!hasVacancy()) {
            throw new IllegalStateException("No hay plazas vacantes. Control de Plazas (Headcount): " +
                    "Impide asignar personal a una posición sin plaza vacante autorizada presupuestariamente.");
        }
        currentSlots++;
    }

    public void vacate() {
        if (currentSlots > 0) {
            currentSlots--;
        }
    }

    public Integer getAvailableSlots() {
        return maxSlots - currentSlots;
    }
}

package com.solveria.core.workforce.domain.model;

import com.solveria.core.workforce.domain.model.vo.Extension;
import com.solveria.core.workforce.domain.model.vo.PartyIdentifierType;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.time.LocalDate;
import java.util.UUID;

/**
 * Entity: PartyIdentifier
 *
 * Representa un documento de identidad (CI, Pasaporte).
 * Tiene ciclo de vida (fechas de emisión y caducidad) y requiere identidad propia.
 *
 * Invariantes:
 * - id_type es requerido
 * - id_number es único
 * - issue_date debe ser anterior a expiry_date
 */
@Getter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class PartyIdentifier {

    private UUID identifierId;
    private UUID personId;
    private PartyIdentifierType idType;
    private String idNumber;
    private Extension extension;
    private LocalDate issueDate;
    private LocalDate expiryDate;

    public static PartyIdentifier create(UUID personId, PartyIdentifierType idType,
                                         String idNumber, Extension extension,
                                         LocalDate issueDate, LocalDate expiryDate) {
        if (idType == null || idNumber == null || idNumber.isBlank()) {
            throw new IllegalArgumentException("idType e idNumber son requeridos");
        }
        if (issueDate != null && expiryDate != null && issueDate.isAfter(expiryDate)) {
            throw new IllegalArgumentException("issueDate no puede ser posterior a expiryDate");
        }

        return PartyIdentifier.builder()
                .identifierId(UUID.randomUUID())
                .personId(personId)
                .idType(idType)
                .idNumber(idNumber)
                .extension(extension)
                .issueDate(issueDate)
                .expiryDate(expiryDate)
                .build();
    }

    public boolean isExpired() {
        if (expiryDate == null) return false;
        return LocalDate.now().isAfter(expiryDate);
    }

    public boolean isExpiringSoon(int daysThreshold) {
        if (expiryDate == null) return false;
        LocalDate thresholdDate = LocalDate.now().plusDays(daysThreshold);
        return LocalDate.now().isBefore(expiryDate) && expiryDate.isBefore(thresholdDate);
    }
}

package com.solveria.core.workforce.domain.model;

import com.solveria.core.workforce.domain.model.vo.ContactPoint;
import com.solveria.core.workforce.domain.model.vo.Gender;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.time.LocalDate;
import java.time.Period;
import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

/**
 * Aggregate Root: Person
 *
 * Identidad civil única del ser humano. Es el anclaje del dominio.
 * Sobrevive a cualquier contrato o relación laboral.
 *
 * Invariantes Críticos:
 * - Identidad Única: Una persona física solo posee un registro maestro (Person ID)
 * - GlobalID es único y sirve para la deduplicación
 * - birth_date >= 18 años al momento del registro
 */
@Getter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Person {

    private UUID personId;
    private String firstName;
    private String lastName;
    private LocalDate birthDate;
    private Gender gender;
    private String globalId;
    private ContactPoint contactPoint;
    private List<PartyIdentifier> identifiers;
    private LocalDate createdAt;
    private LocalDate updatedAt;

    private static final int MINIMUM_AGE = 18;

    public static Person create(String firstName, String lastName, LocalDate birthDate,
                                Gender gender, String globalId, ContactPoint contactPoint) {
        // Validar age >= 18
        Period age = Period.between(birthDate, LocalDate.now());
        if (age.getYears() < MINIMUM_AGE) {
            throw new IllegalArgumentException("La persona debe tener mínimo " + MINIMUM_AGE + " años");
        }

        if (firstName == null || firstName.isBlank() ||
            lastName == null || lastName.isBlank() ||
            globalId == null || globalId.isBlank()) {
            throw new IllegalArgumentException("firstName, lastName y globalId son requeridos");
        }

        return Person.builder()
                .personId(UUID.randomUUID())
                .firstName(firstName)
                .lastName(lastName)
                .birthDate(birthDate)
                .gender(gender)
                .globalId(globalId)
                .contactPoint(contactPoint != null ? contactPoint : new ContactPoint())
                .identifiers(new ArrayList<>())
                .createdAt(LocalDate.now())
                .updatedAt(LocalDate.now())
                .build();
    }

    public void addIdentifier(PartyIdentifier identifier) {
        if (identifier == null) {
            throw new IllegalArgumentException("identifier no puede ser nulo");
        }
        // Evitar duplicados por idNumber
        boolean exists = identifiers.stream()
                .anyMatch(id -> id.getIdNumber().equals(identifier.getIdNumber()));
        if (exists) {
            throw new IllegalArgumentException("El identificador " + identifier.getIdNumber() +
                    " ya está vinculado a esta persona");
        }
        identifiers.add(identifier);
    }

    public void updateContactPoint(ContactPoint newContactPoint) {
        this.contactPoint = newContactPoint;
        this.updatedAt = LocalDate.now();
    }

    public Integer getAge() {
        return Period.between(birthDate, LocalDate.now()).getYears();
    }

    public String getFullName() {
        return firstName + " " + lastName;
    }
}

package com.solveria.core.workforce.domain.model;

import com.solveria.core.workforce.domain.model.vo.HeadcountPlan;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.util.UUID;

/**
 * Aggregate Root: Position
 *
 * Representa la "silla" física o virtual en la organización.
 * Existe independientemente de si hay alguien ocupándola.
 *
 * Invariantes:
 * - Control de Plazas (Headcount): No se puede exceder el límite de slots
 */
@Getter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Position {

    private UUID positionId;
    private UUID unitId;
    private UUID jobId;
    private PositionStatus status;
    private Boolean isBudgeted;
    private HeadcountPlan headcountPlan;

    public enum PositionStatus {
        VACANT("Vacante"),
        OCCUPIED("Ocupada"),
        RESERVED("Reservada");

        private final String label;

        PositionStatus(String label) {
            this.label = label;
        }

        public String getLabel() {
            return label;
        }
    }

    public static Position create(UUID unitId, UUID jobId, Boolean isBudgeted, Integer maxSlots) {
        if (unitId == null || jobId == null || maxSlots == null) {
            throw new IllegalArgumentException("unitId, jobId y maxSlots son requeridos");
        }
        if (maxSlots <= 0) {
            throw new IllegalArgumentException("maxSlots debe ser mayor a 0");
        }

        return Position.builder()
                .positionId(UUID.randomUUID())
                .unitId(unitId)
                .jobId(jobId)
                .status(PositionStatus.VACANT)
                .isBudgeted(isBudgeted != null ? isBudgeted : false)
                .headcountPlan(HeadcountPlan.create(maxSlots))
                .build();
    }

    public void occupy() {
        if (!PositionStatus.VACANT.equals(status)) {
            throw new IllegalStateException("Solo se puede ocupar una posición vacante");
        }
        headcountPlan.occupy();
        this.status = PositionStatus.OCCUPIED;
    }

    public void vacate() {
        headcountPlan.vacate();
        if (headcountPlan.getCurrentSlots() == 0) {
            this.status = PositionStatus.VACANT;
        }
    }

    public void reserve() {
        if (!PositionStatus.VACANT.equals(status)) {
            throw new IllegalStateException("Solo se puede reservar una posición vacante");
        }
        this.status = PositionStatus.RESERVED;
    }

    public boolean hasVacancy() {
        return headcountPlan.hasVacancy();
    }
}

package com.solveria.core.workforce.domain.model;

import lombok.Getter;
import lombok.NoArgsConstructor;

import java.time.LocalDate;
import java.util.UUID;

/**
 * Aggregate Root: Relationship
 *
 * Vínculo legal entre la Person y el Tenant.
 * Define si el perfil será WorkerProfile o AcademicProfile.
 * Soporta el "Multi-vínculo" (ej. administrativo y docente simultáneo).
 *
 * Invariantes Críticos:
 * - No Traslape de Vínculos Primarios: Impide dos vínculos laborales primarios
 *   de tiempo completo activos para evitar contingencias legales por doble percepción
 */
@Getter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Relationship {

    private UUID relationshipId;
    private UUID personId;
    private UUID tenantId;
    private RelationshipType relationType;
    private RelationshipStatus currentStatus;
    private LocalDate hireDate;
    private LocalDate createdAt;
    private LocalDate updatedAt;

    public enum RelationshipType {
        LABOR("Laboral"),
        ACADEMIC("Académico"),
        INTERNSHIP("Pasantía");

        private final String label;

        RelationshipType(String label) {
            this.label = label;
        }

        public String getLabel() {
            return label;
        }
    }

    public enum RelationshipStatus {
        DRAFT("Draft"),
        ACTIVE("Activo"),
        SUSPENDED("Suspendido"),
        TERMINATED("Terminado");

        private final String label;

        RelationshipStatus(String label) {
            this.label = label;
        }

        public String getLabel() {
            return label;
        }
    }

    public static Relationship create(UUID personId, UUID tenantId, RelationshipType relationType,
                                     LocalDate hireDate) {
        if (personId == null || tenantId == null || relationType == null) {
            throw new IllegalArgumentException("personId, tenantId y relationType son requeridos");
        }

        return Relationship.builder()
                .relationshipId(UUID.randomUUID())
                .personId(personId)
                .tenantId(tenantId)
                .relationType(relationType)
                .currentStatus(RelationshipStatus.DRAFT)
                .hireDate(hireDate)
                .createdAt(LocalDate.now())
                .updatedAt(LocalDate.now())
                .build();
    }

    public void activate() {
        if (!RelationshipStatus.DRAFT.equals(currentStatus)) {
            throw new IllegalStateException("Solo se puede activar una relación en estado DRAFT");
        }
        this.currentStatus = RelationshipStatus.ACTIVE;
        this.updatedAt = LocalDate.now();
    }

    public void suspend() {
        this.currentStatus = RelationshipStatus.SUSPENDED;
        this.updatedAt = LocalDate.now();
    }

    public void reactivate() {
        if (!RelationshipStatus.SUSPENDED.equals(currentStatus)) {
            throw new IllegalStateException("Solo se puede reactivar una relación SUSPENDIDA");
        }
        this.currentStatus = RelationshipStatus.ACTIVE;
        this.updatedAt = LocalDate.now();
    }

    public void terminate() {
        this.currentStatus = RelationshipStatus.TERMINATED;
        this.updatedAt = LocalDate.now();
    }
}

        ADJUNTO("Adjunto"),

        INVESTIGADOR("Investigador");

        private final String label;

                .academicId(UUID.randomUUID())
                .relationshipId(relationshipId)
                .currentRank(initialRank)
                .teachingLoad(teachingLoad)
                .build();
    }

        if (newRank == null) {
            throw new IllegalArgumentException("newRank no puede ser nulo");
        }
        this.currentRank = newRank;
    }

    public void updateTeachingLoad(Integer newLoad) {
        if (newLoad == null || newLoad <= 0) {
            throw new IllegalArgumentException("Teaching load debe ser mayor a 0");
        }
        this.teachingLoad = newLoad;
    }

// Ruta: core-platform/src/main/java/com/solveria/core/workforce/infrastructure/outbox/EventOutbox.java
package com.solveria.core.workforce.infrastructure.outbox;

import lombok.*;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.UUID;

/**
 * EventOutbox - Patrón Outbox
 *
 * Tabla de eventos que garantiza la consistencia eventual.
 * Toda acción de dominio guarda en la misma transacción JPA:
 * 1. El estado de la entidad
 * 2. Un evento en la tabla event_outbox (con campo JSONB para el payload)
 *
 * Un Scheduler interno en Spring Boot leerá esta tabla y despachará
 * eventos nativamente mediante ApplicationEventPublisher.
 */
@Entity
@Table(name = "event_outbox", indexes = {
        @Index(name = "idx_outbox_published", columnList = "is_published"),
        @Index(name = "idx_outbox_created", columnList = "created_at")
})
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class EventOutbox {

    @Id
    private UUID eventId;

    @Column(name = "aggregate_type", nullable = false)
    private String aggregateType;  // ej: "Person", "OrgUnit", "Relationship"

    @Column(name = "aggregate_id", nullable = false)
    private UUID aggregateId;

    @Column(name = "event_type", nullable = false)
    private String eventType;  // ej: "PERSON_CREATED", "ORG_UNIT_CREATED"

    @Column(name = "payload", columnDefinition = "JSONB", nullable = false)
    private String payload;  // JSON serializado del evento

    @Column(name = "is_published", nullable = false)
    private Boolean isPublished;

    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;

    @Column(name = "published_at")
    private LocalDateTime publishedAt;

    public static EventOutbox create(String aggregateType, UUID aggregateId,
                                    String eventType, String payload) {
        return EventOutbox.builder()
                .eventId(UUID.randomUUID())
                .aggregateType(aggregateType)
                .aggregateId(aggregateId)
                .eventType(eventType)
                .payload(payload)
                .isPublished(false)
                .createdAt(LocalDateTime.now())
                .build();
    }

    public void markAsPublished() {
        this.isPublished = true;
        this.publishedAt = LocalDateTime.now();
    }
}

 * - Identidad Única: Deduplicación por CI/Pasaporte

    private final PartyIdentifierRepository identifierRepository;
    private final EventOutboxRepository eventOutboxRepository;

                request.getGlobalId(),

/**
 * Use Case: Create OrgUnit
 *
 * Crea nodos en la jerarquía organizacional.
 * Valida que no existan unidades huérfanas.
 *
 * Validación de Seguridad Multi-Tenant (SoD):
 * - Obtiene tenantId del contexto de seguridad
 * - Valida que el parent pertenezca al mismo tenant
 */
@Slf4j
@Service
@RequiredArgsConstructor
@Transactional

    private final OrgUnitRepository orgUnitRepository;
    private final EventOutboxRepository eventOutboxRepository;
    private final OrgUnitMapper orgUnitMapper;

    public OrgUnitResponse executeRoot(CreateOrgUnitRequest request) {
        // Obtener tenant desde el contexto de seguridad
        UUID tenantId = UUID.fromString(SecurityTenantContext.getTenantId());

        log.info("event=ORG_UNIT_ROOT_CREATE_REQUEST_RECEIVED name={} tenantId={}",
                request.getName(), tenantId);

        CostCenter costCenter = CostCenter.create(request.getCostCode(), request.getCostDescription());

        OrgUnit.OrgUnitType unitType = OrgUnit.OrgUnitType.valueOf(request.getUnitType().toUpperCase());

        OrgUnit orgUnit = OrgUnit.createRoot(tenantId, request.getName(), unitType, costCenter);

        OrgUnit savedOrgUnit = orgUnitRepository.save(orgUnitMapper.toPersistenceEntity(orgUnit));

        // Generar evento para Outbox
        String eventPayload = orgUnitMapper.toEventPayload(savedOrgUnit);
        EventOutbox outboxEvent = EventOutbox.create(
                "OrgUnit",
                savedOrgUnit.getUnitId(),
                "ORG_UNIT_CREATED",
                eventPayload
        );
        eventOutboxRepository.save(outboxEvent);

        log.info("event=ORG_UNIT_ROOT_CREATE_SUCCESS unitId={} name={}",
                savedOrgUnit.getUnitId(), savedOrgUnit.getName());

        return orgUnitMapper.toResponse(savedOrgUnit);
    }

    public OrgUnitResponse executeChild(CreateOrgUnitRequest request, UUID parentId) {
        // Obtener tenant desde el contexto de seguridad
        UUID tenantId = UUID.fromString(SecurityTenantContext.getTenantId());

        log.info("event=ORG_UNIT_CHILD_CREATE_REQUEST_RECEIVED name={} parentId={} tenantId={}",
                request.getName(), parentId, tenantId);

        // Validación: Parent existe Y pertenece al tenant actual
        // Esto previene que un usuario de un tenant acceda a la jerarquía de otro tenant
        var parentOrgUnit = orgUnitRepository.findByUnitIdAndTenantId(parentId, tenantId)
                .orElseThrow(() -> {
                    log.warn("event=ORG_UNIT_SECURITY_VIOLATION parentId={} tenantId={}",
                        parentId, tenantId);
                    return new OrgUnitNotFoundException("OrgUnit padre no encontrado o no pertenece a este tenant");
                });

        // Invariante: No Unidades Huérfanas
        if (parentOrgUnit == null) {
            throw new IllegalStateException("Invariante violada: No Unidades Huérfanas");
        }

        CostCenter costCenter = CostCenter.create(request.getCostCode(), request.getCostDescription());

        OrgUnit.OrgUnitType unitType = OrgUnit.OrgUnitType.valueOf(request.getUnitType().toUpperCase());

        OrgUnit orgUnit = OrgUnit.createChild(tenantId, parentId, request.getName(), unitType, costCenter);

        OrgUnit savedOrgUnit = orgUnitRepository.save(orgUnitMapper.toPersistenceEntity(orgUnit));

        // Generar evento para Outbox
        String eventPayload = orgUnitMapper.toEventPayload(savedOrgUnit);
        EventOutbox outboxEvent = EventOutbox.create(
                "OrgUnit",
                savedOrgUnit.getUnitId(),
                "ORG_UNIT_CREATED",
                eventPayload
        );
        eventOutboxRepository.save(outboxEvent);

        log.info("event=ORG_UNIT_CHILD_CREATE_SUCCESS unitId={} name={} parentId={}",
                savedOrgUnit.getUnitId(), savedOrgUnit.getName(), parentId);

        return orgUnitMapper.toResponse(savedOrgUnit);
    }
}

package com.solveria.core.workforce.application.usecase;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * Use Case: Create Relationship
 *
 * Crea el vínculo legal entre la Person y el Tenant.
 * Valida la Invariante: No Traslape de Vínculos Primarios
 */
@Slf4j
@Service
@RequiredArgsConstructor
@Transactional
public class CreateRelationshipUseCase {

    private final RelationshipRepository relationshipRepository;
    private final PersonRepository personRepository;
    private final WorkerProfileRepository workerProfileRepository;
    private final AcademicProfileRepository academicProfileRepository;
    private final EventOutboxRepository eventOutboxRepository;
    private final RelationshipMapper relationshipMapper;

    public RelationshipResponse execute(CreateRelationshipRequest request) {
        log.info("event=RELATIONSHIP_CREATE_REQUEST_RECEIVED personId={} tenantId={} type={}",
                request.getPersonId(), request.getTenantId(), request.getRelationType());

        // Validación: Person existe
        boolean personExists = personRepository.existsById(request.getPersonId());
        if (!personExists) {
            throw new PersonNotFoundException("Person no encontrado");
        }

        // Invariante: No Traslape de Vínculos Primarios
        Relationship.RelationshipType relType = Relationship.RelationshipType.valueOf(
                request.getRelationType().toUpperCase()
        );

        if (Relationship.RelationshipType.LABOR.equals(relType)) {
            // Buscar vínculos laborales activos primarios
            boolean hasPrimaryLabor = relationshipRepository
                    .existsByPersonIdAndTenantIdAndRelationTypeAndCurrentStatus(
                            request.getPersonId(),
                            request.getTenantId(),
                            Relationship.RelationshipType.LABOR,
                            Relationship.RelationshipStatus.ACTIVE
                    );

            if (hasPrimaryLabor) {
                throw new IllegalStateException(
                        "Invariante violada: Un colaborador no puede tener dos jefes administrativos simultáneos"
                );
            }
        }

        // Crear entidad Relationship
        Relationship relationship = Relationship.create(
                request.getPersonId(),
                request.getTenantId(),
                relType,
                request.getHireDate()
        );

        Relationship savedRelationship = relationshipRepository
                .save(relationshipMapper.toPersistenceEntity(relationship));

        // Crear perfil específico (WorkerProfile o AcademicProfile)
        if (Relationship.RelationshipType.ACADEMIC.equals(relType)) {
            AcademicProfile academicProfile = AcademicProfile.create(
                    savedRelationship.getRelationshipId(),
                    AcademicProfile.AcademicRank.AUXILIAR,
                    request.getTeachingLoad() != null ? request.getTeachingLoad() : 0
            );
            academicProfileRepository.save(academicProfile);
        } else {
            WorkerProfile workerProfile = WorkerProfile.create(
                    savedRelationship.getRelationshipId(),
                    request.getEmployeeNo(),
                    request.getDepartment(),
                    request.getJobTitle()
            );
            workerProfileRepository.save(workerProfile);
        }

        // Generar evento para Outbox
        String eventPayload = relationshipMapper.toEventPayload(savedRelationship);
        EventOutbox outboxEvent = EventOutbox.create(
                "Relationship",
                savedRelationship.getRelationshipId(),
                "RELATIONSHIP_CREATED",
                eventPayload
        );
        eventOutboxRepository.save(outboxEvent);

        log.info("event=RELATIONSHIP_CREATE_SUCCESS relationshipId={} personId={}",
                savedRelationship.getRelationshipId(), savedRelationship.getPersonId());

        return relationshipMapper.toResponse(savedRelationship);
    }
}

                    "PERSON_CREATE_FAILED", e.getMessage(), e);
