// Ruta: core-platform/src/main/java/com/solveria/core/workforce/infrastructure/adapter/PersonMapper.java
package com.solveria.core.workforce.infrastructure.adapter;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.solveria.core.workforce.domain.model.Person;
import com.solveria.core.workforce.infrastructure.jpa.PersonJpa;
import com.solveria.core.workforce.application.dto.PersonResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

import java.util.LinkedHashMap;
import java.util.Map;

/**
 * Mapper: Domain Person ↔ JPA PersonJpa ↔ DTO PersonResponse
 *
 * Responsable de convertir entre capas:
 * - Domain Model → JPA Entity (persistencia)
 * - JPA Entity → Domain Model (lectura)
 * - Domain Model → DTO Response (API REST)
 * - Domain Model → Event Outbox Payload (eventos)
 */
@Component
@RequiredArgsConstructor
public class PersonMapper {

    private final ObjectMapper objectMapper;

    /**
     * Convierte Domain Person a JPA PersonJpa para persistencia
     */
    public PersonJpa toPersistenceEntity(Person person) {
        if (person == null) return null;

        return PersonJpa.builder()
                .personId(person.getPersonId())
                .firstName(person.getFirstName())
                .lastName(person.getLastName())
                .birthDate(person.getBirthDate())
                .gender(person.getGender())
                .globalId(person.getGlobalId())
                .contactPoint(person.getContactPoint())
                .createdAt(person.getCreatedAt())
                .updatedAt(person.getUpdatedAt())
                .build();
    }

    /**
     * Convierte JPA PersonJpa a Domain Person
     */
    public Person toDomain(PersonJpa jpa) {
        if (jpa == null) return null;

        return Person.builder()
                .personId(jpa.getPersonId())
                .firstName(jpa.getFirstName())
                .lastName(jpa.getLastName())
                .birthDate(jpa.getBirthDate())
                .gender(jpa.getGender())
                .globalId(jpa.getGlobalId())
                .contactPoint(jpa.getContactPoint())
                .createdAt(jpa.getCreatedAt())
                .updatedAt(jpa.getUpdatedAt())
                .build();
    }

    /**
     * Convierte Domain Person a DTO PersonResponse para REST
     */
    public PersonResponse toResponse(Person person) {
        if (person == null) return null;

        return PersonResponse.builder()
                .personId(person.getPersonId())
                .firstName(person.getFirstName())
                .lastName(person.getLastName())
                .birthDate(person.getBirthDate())
                .gender(person.getGender().name())
                .globalId(person.getGlobalId())
                .age(person.getAge())
                .email(person.getContactPoint() != null ? person.getContactPoint().getEmail() : null)
                .phone(person.getContactPoint() != null ? person.getContactPoint().getPhone() : null)
                .address(person.getContactPoint() != null ? person.getContactPoint().getAddress() : null)
                .createdAt(person.getCreatedAt())
                .build();
    }

    /**
     * Convierte Domain Person a JSON payload para Outbox
     */
    public String toEventPayload(Person person) {
        if (person == null) return "{}";

        try {
            Map<String, Object> payload = new LinkedHashMap<>();
            payload.put("personId", person.getPersonId());
            payload.put("firstName", person.getFirstName());
            payload.put("lastName", person.getLastName());
            payload.put("birthDate", person.getBirthDate().toString());
            payload.put("gender", person.getGender().name());
            payload.put("globalId", person.getGlobalId());
            payload.put("age", person.getAge());

            return objectMapper.writeValueAsString(payload);
        } catch (Exception e) {
            throw new RuntimeException("Error serializando Person a JSON", e);
        }
    }
}

package com.solveria.core.workforce.domain.exception;

/**
 * Excepción base del dominio
 *
 * Todas las excepciones del core heredan de esta clase.
 * Permite manejo centralizado y coherente de errores.
 */
public class SolverException extends RuntimeException {

    private static final long serialVersionUID = 1L;
    private final String errorCode;

    public SolverException(String errorCode) {
        super(errorCode);
        this.errorCode = errorCode;
    }

    public SolverException(String errorCode, String message) {
        super(message);
        this.errorCode = errorCode;
    }

    public SolverException(String errorCode, String message, Throwable cause) {
        super(message, cause);
        this.errorCode = errorCode;
    }

    public String getErrorCode() {
        return errorCode;
    }
}
